def Essence(a):
	if a == 0:
		return 'Error'
	else:
		return ((1.10 * a) + (0.36 * a)) * 6

		
		
def Vivify (b):
	if b == 0:
		return 'Error'
	else:
		return ((b * 2.75) * 3)

		
		
def Compare (c):
	if c == 0:
		return 'Error'
	else:
		return Essence (c) - Vivify (c)
